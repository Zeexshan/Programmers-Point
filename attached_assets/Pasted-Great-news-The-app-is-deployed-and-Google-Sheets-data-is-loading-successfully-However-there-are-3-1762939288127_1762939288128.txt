### **In `client/src/pages/admin/Inquiries.tsx`:**

**Add "Mark as Joined" functionality:**

```typescript
async function updateInquiryStatus(inquiryIndex: number, newStatus: 'Pending' | 'Joined') {
  try {
    setLoading(true);
    
    // Use Google Apps Script to update status
    const APPS_SCRIPT_URL = import.meta.env.VITE_APPS_SCRIPT_URL;
    
    await fetch(APPS_SCRIPT_URL, {
      method: 'POST',
      mode: 'no-cors',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        action: 'updateInquiryStatus',
        rowIndex: inquiryIndex + 2, // +2 because row 1 is header, array is 0-indexed
        status: newStatus
      })
    });

    // Refresh data
    await loadInquiries();
    
    toast({
      title: 'Success',
      description: `Inquiry marked as ${newStatus}`,
    });
  } catch (error) {
    toast({
      title: 'Error',
      description: 'Failed to update status',
      variant: 'destructive'
    });
  } finally {
    setLoading(false);
  }
}
```
